    keyboard_configuration.json
    mouse_configuration.json

    "description": "An example configuration file for an archer",
    "maxHealth": 15,
    "strength": 0,
    "rangedStrength": 10,
    "speed": 1,
    "range": 2,
    "areaOfEffect": "Skewer",
    "attributes": []
    }
